MACRO(ALTER nm text)
  IF(${nm})
    MESSAGE(STATUS "Altering ${text} to ${${nm}}")
    TARGET_COMPILE_DEFINITIONS(Esprit_impl PRIVATE ${nm}=${${nm}})
  ENDIF()
ENDMACRO()
